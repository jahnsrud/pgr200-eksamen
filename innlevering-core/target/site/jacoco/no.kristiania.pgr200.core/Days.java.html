package no.kristiania.pgr200.core;

import java.sql.Date;

public class Days {

	private int dayId;
	private int timeslotId;
	private Date dateOfConference;
	private int id;

	public Days(int dayId, int timeslotId, Date dateOfConference, int id) {
		this.dayId = dayId;
		this.timeslotId = timeslotId;
		this.dateOfConference = dateOfConference;
		this.id = id;
	}

	public int getDayId() {
		return dayId;
	}

	public void setDayId(int dayId) {
		this.dayId = dayId;
	}

	public int getTimeslotId() {
		return timeslotId;
	}

	public void setTimeslotId(int timeslotId) {
		this.timeslotId = timeslotId;
	}

	public Date getDateOfConference() {
		return dateOfConference;
	}

	public void setDateOfConference(Date dateOfConference) {
		this.dateOfConference = dateOfConference;
	}

	public int getId() {
		return id;
	}

	public void setId(int id) {
		this.id = id;
	}
}
